59 bool FrameConverter::ConvertFrame(const AVFrame* frame,
60  const uint8_t* const** data,
61  const int** linesize,
62  AVPixelFormat desired_pixel_format) {
63  if (IsHardwarePixelFormat(static_cast<AVPixelFormat>(frame->format))) {
64  if (!cpu_frame_) {
65  cpu_frame_ = av_frame_alloc();
66  if (!cpu_frame_) {
67  LOG(ERROR) << "Error allocating frame for conversion";
68  return false;
69  }
70  }
71 
72  av_frame_unref(cpu_frame_);
73  const int copy_code = av_hwframe_transfer_data(cpu_frame_, frame, 0);
74  if (copy_code < 0) {
75  LOG(ERROR) << "Error transferring frame data to CPU: " << copy_code;
76  return false;
77  }
78  frame = cpu_frame_;
79  }
80 
81  if (frame->format == desired_pixel_format) {
82  *data = frame->data;
83  *linesize = frame->linesize;
84  return true;
85  }
86 
87 #ifdef HAS_SWSCALE
88  if (frame->width != convert_frame_width_ ||
89  frame->height != convert_frame_height_ ||
90  desired_pixel_format != convert_pixel_format_) {
91  av_freep(&convert_frame_data_[0]);
92  if (av_image_alloc(convert_frame_data_, convert_frame_linesize_,
93  frame->width, frame->height, desired_pixel_format,
94  16) < 0) {
95  LOG(ERROR) << "Error allocating frame for conversion";
96  return false;
97  }
98  convert_frame_width_ = frame->width;
99  convert_frame_height_ = frame->height;
100  convert_pixel_format_ = desired_pixel_format;
101  }
102 
103  sws_ctx_ = sws_getCachedContext(
104  sws_ctx_, frame->width, frame->height,
105  static_cast<AVPixelFormat>(frame->format), frame->width, frame->height,
106  desired_pixel_format, 0, nullptr, nullptr, nullptr);
107  if (!sws_ctx_) {
108  LOG(ERROR) << "Error allocating conversion context";
109  return false;
110  }
111 
112  sws_scale(sws_ctx_, frame->data, frame->linesize, 0, frame->height,
113  convert_frame_data_, convert_frame_linesize_);
114 
115  *data = convert_frame_data_;
116  *linesize = convert_frame_linesize_;
117  return true;
118 #else
119  LOG_ONCE(INFO) << "Not built to convert pixel formats";
120  return false;
121 #endif
122 }
