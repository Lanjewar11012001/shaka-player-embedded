60 void Video::Initialize(Client* client, media::VideoRenderer* video_renderer,
61  media::AudioRenderer* audio_renderer) {
62  // This can be called immediately after the JsManager constructor. Since the
63  // Environment might not be setup yet, run this in an internal task so we know
64  // it is ready.
65  const auto callback = [=]() {
66  impl_->inner = new js::mse::HTMLVideoElement(
67  js::dom::Document::GetGlobalDocument(), video_renderer, audio_renderer);
68 
69  if (client) {
70  impl_->inner->SetCppEventListener(js::EventType::Playing,
71  std::bind(&Client::OnPlaying, client));
72  impl_->inner->SetCppEventListener(js::EventType::Pause,
73  std::bind(&Client::OnPause, client));
74  impl_->inner->SetCppEventListener(js::EventType::Ended,
75  std::bind(&Client::OnEnded, client));
76  impl_->inner->SetCppEventListener(js::EventType::Seeking,
77  std::bind(&Client::OnSeeking, client));
78  impl_->inner->SetCppEventListener(js::EventType::Seeked,
79  std::bind(&Client::OnSeeked, client));
80  }
81  };
82  JsManagerImpl::Instance()
83  ->MainThread()
84  ->AddInternalTask(TaskPriority::Internal, "Video init",
85  PlainCallbackTask(callback))
86  ->GetValue();
87 }
