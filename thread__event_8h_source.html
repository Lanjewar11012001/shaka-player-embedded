143  bool SignalAllIfNotSet() {
144  // Don't signal with the lock held since once we signal, this object may be
145  // destroyed by thread we woke up.
146  std::promise<void> p;
147  {
148  std::unique_lock<std::mutex> lock(mutex_);
149  if (is_set_)
