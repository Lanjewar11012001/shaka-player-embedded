45 class SupportCache {
46  public:
47  bool TryGet(const std::string& codec, int width, int height, bool* result) {
48  std::unique_lock<std::mutex> lock(mutex_);
49  CheckArgs args(codec, width, height);
50  if (results_.count(args) > 0) {
51  *result = results_.at(args);
52  return true;
53  } else {
54  return false;
55  }
56  }
57 
58  void Insert(const std::string& codec, int width, int height, bool result) {
59  std::unique_lock<std::mutex> lock(mutex_);
60  results_[CheckArgs(codec, width, height)] = result;
61  }
62 
63  private:
64  struct CheckArgs {
65  CheckArgs(const std::string& codec, int width, int height)
66  : codec(codec), width(width), height(height) {}
67 
68  const std::string codec;
69  const int width;
70  const int height;
71 
72  bool operator==(const CheckArgs& other) const noexcept {
73  return codec == other.codec && width == other.width &&
74  height == other.height;
75  }
76  };
77  struct CheckArgsHash {
78  size_t operator()(const CheckArgs& args) const noexcept {
79  return std::hash<std::string>()(args.codec) ^ (args.width << 16) ^
80  args.height;
81  }
82  };
83 
84  std::mutex mutex_;
85  std::unordered_map<CheckArgs, bool, /* Hash */ CheckArgsHash> results_;
86 };

90 util::CFRef<CFMutableDictionaryRef> GetExtraData(CMVideoCodecType codec,
91  int profile) {
92  util::CFRef<CFMutableDictionaryRef> avc_info = CFDictionaryCreateMutable(
93  kCFAllocatorDefault, 1, &kCFTypeDictionaryKeyCallBacks,
94  &kCFTypeDictionaryValueCallBacks);
95 
96  switch (codec) {
97  case kCMVideoCodecType_H264: {
98  // This is just a common SPS and PPS that doesn't use any "unusual"
99  // features; this is believed to be commonly supported. We can't just
100  // pass 0 SPS or PPS, the decoder requires at least one of each.
101  uint8_t extra_data[] = {
102  // Disable formatting so clang-format doesn't put each element on its
103  // own line.
104  // clang-format off
105  0x01, // version
106  (profile >> 16) & 0xff, // profile
107  (profile >> 8) & 0xff, // profile compat
108  (profile >> 0) & 0xff, // level
109  0xff, // 6 reserved bits + 2 bits nalu size length - 1
110 
111  0xe1, // 3 reserved bits + 5 bits SPS count
112  0x00, 0x1c, // SPS size
113  0x67, 0x42, 0xc8, 0x1e, 0xd9, 0x01, 0x03, 0xfe, 0xbf, 0xf0,
114  0x06, 0xe0, 0x06, 0xd1, 0x00, 0x00, 0x03, 0x00, 0x01, 0x00,
115  0x00, 0x03, 0x00, 0x30, 0x0f, 0x16, 0x2e, 0x48,
116 
117  0x01, // PPS count
118  0x00, 0x04, // PPS size
119  0x68, 0xcb, 0x8c, 0xb2,
120  // clang-format on
121  };
122  util::CFRef<CFDataRef> buffer =
123  CFDataCreate(kCFAllocatorDefault, extra_data, sizeof(extra_data));
124  if (buffer)
125  CFDictionarySetValue(avc_info, CFSTR("avcC"), buffer);
126  break;
127  }
128  default:
129  LOG(FATAL) << "Unknown codec type: " << codec;
130  }
131 
132  return avc_info;
133 }

138 bool IosHardwareSupport(CMVideoCodecType codec, int profile, int width,
139  int height) {
140  if (width * height > 5000000) {
141  // VideoToolbox doesn't handle out of memory correctly and has a tendency to
142  // just crash with a memory error if we run out of memory. This only seems
143  // to happen with 4k, so just blacklist it for now.
144  // TODO: Find a better solution or file a bug.
145  return false;
146  }
147 
148  auto NewDict = []() {
149  return CFDictionaryCreateMutable(kCFAllocatorDefault, 0,
150  &kCFTypeDictionaryKeyCallBacks,
151  &kCFTypeDictionaryValueCallBacks);
152  };
153 
154  util::CFRef<CFMutableDictionaryRef> decoder_config = NewDict();
155  CFDictionarySetValue(
156  decoder_config,
157  kVTVideoDecoderSpecification_RequireHardwareAcceleratedVideoDecoder,
158  kCFBooleanTrue);
159  CFDictionarySetValue(
160  decoder_config,
161  kCMFormatDescriptionExtension_SampleDescriptionExtensionAtoms,
162  GetExtraData(codec, profile));
163 
164  CMFormatDescriptionRef cm_fmt_desc;
165  if (CMVideoFormatDescriptionCreate(kCFAllocatorDefault, codec, width, height,
166  decoder_config, &cm_fmt_desc) != 0) {
167  return false;
168  }
169 
170  // Create a dictionary of buffer properties.
171  util::CFRef<CFNumberRef> w =
172  CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt32Type, &width);
173  util::CFRef<CFNumberRef> h =
174  CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt32Type, &height);
175  const OSType pix_fmt = kCVPixelFormatType_420YpCbCr8BiPlanarVideoRange;
176  util::CFRef<CFNumberRef> cv_pix_fmt =
177  CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt32Type, &pix_fmt);
178  util::CFRef<CFMutableDictionaryRef> buffer_attributes = NewDict();
179  util::CFRef<CFMutableDictionaryRef> io_surface_properties = NewDict();
180  CFDictionarySetValue(buffer_attributes, kCVPixelBufferPixelFormatTypeKey,
181  cv_pix_fmt);
182  CFDictionarySetValue(buffer_attributes, kCVPixelBufferIOSurfacePropertiesKey,
183  io_surface_properties);
184  CFDictionarySetValue(buffer_attributes, kCVPixelBufferWidthKey, w);
185  CFDictionarySetValue(buffer_attributes, kCVPixelBufferHeightKey, h);
186 
187 
188  // Try to create the decompression session, which will tell us whether it
189  // supports these settings.
190  VTDecompressionOutputCallbackRecord decoder_cb = {
191  .decompressionOutputCallback = &IosDecoderCallback,
192  .decompressionOutputRefCon = nullptr,
193  };
194  VTDecompressionSessionRef session = nullptr;
195  const OSStatus status =
196  VTDecompressionSessionCreate(nullptr, cm_fmt_desc, decoder_config,
197  buffer_attributes, &decoder_cb, &session);
198  CFRelease(cm_fmt_desc);
199  if (session)
200  CFRelease(session);
201 
202  switch (status) {
203  case 0:
204  return true;
205 
206  case kVTCouldNotFindVideoDecoderErr:
207  case kVTVideoDecoderNotAvailableNowErr:
208  LOG(INFO) << "Hardware decoder not available";
209  return false;
210  case kVTVideoDecoderUnsupportedDataFormatErr:
211  VLOG(1) << "Video not supported: size=" << width << "x" << height
212  << ", profile=" << std::hex << profile;
213  return false;
214  default:
215  LOG(DFATAL) << "Bad hardware acceleration query: status=" << status;
216  return false;
217  }
218 }

222 bool InternalHardwareSupport(const std::string& codec, int width, int height) {
223 #ifdef OS_IOS
224  if (codec.substr(0, 5) == "avc1.") {
225  long profile = strtol(codec.substr(5).c_str(), nullptr, 16); // NOLINT
226  if (profile == 0)
227  profile = 0x42001e;
228  return IosHardwareSupport(kCMVideoCodecType_H264, profile, width, height);
229  } else if (codec.substr(0, 5) == "mp4a.") {
230  return true;
231  } else {
232  LOG(ERROR) << "Unable to query support for codec: " << codec;
233  return false;
234  }
235 #else
236 # error "This platform doesn't have a unique hardware decoder to query."
237 #endif
238 }

242 bool DoesHardwareSupportCodec(const std::string& codec, int width, int height) {
243  if (width == 0)
244  width = 720;
245  if (height == 0)
246  height = 480;
247 
248  BEGIN_ALLOW_COMPLEX_STATICS
249  static SupportCache cache_;
250  END_ALLOW_COMPLEX_STATICS
251  bool result;
252  if (!cache_.TryGet(codec, width, height, &result)) {
253  result = InternalHardwareSupport(codec, width, height);
254  cache_.Insert(codec, width, height, result);
255  }
256  return result;
257 }
